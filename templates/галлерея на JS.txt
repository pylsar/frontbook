var btn_prev = document.querySelector("#gallery .buttons .back");
var btn_next = document.querySelector("#gallery .buttons .future");
var images = document.querySelectorAll("#gallery .photo img")
var i = 0; // номер текущей картинки

btn_prev.onclick = function(){
	images[i].style.display = "none";
	i--;
	if(i < 0) {
		i = images.length - 1;
	}
	images[i].style.display = "block";
}
btn_next.onclick = function(){
	images[i].style.display = "none";
	i++; 
	if(i >= images.length) {
		i=0;
	}
	images[i].style.display = "block";
}
